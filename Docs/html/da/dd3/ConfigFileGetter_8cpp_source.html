364 std::string ConfigFileGetter::GetFileName(const std::string &path)
365 {
366  Library::Utils::Logger &logger = Library::Utils::Logger::GetInstance();
367  size_t lastSlashPos = path.find_last_of('/');
368 
369  if (lastSlashPos == std::string::npos)
370  {
371  logger.WriteError("Error: Invalid path format");
372  return "";
373  }
374  size_t cfgPos = path.find_last_of('.');
375  if (cfgPos == std::string::npos || cfgPos <= lastSlashPos)
376  {
377  logger.WriteError("Error: Invalid path format");
378  return "";
379  }
380  return path.substr(lastSlashPos + 1, cfgPos - lastSlashPos - 1);
381 }
