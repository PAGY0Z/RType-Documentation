99 void Spritesheet::UpdateFrame(const std::vector<int> &frames, int tile_width, int tile_height, int size_width, int size_height)
100 {
101  for (auto frame : frames)
102  {
103  if (__frames.find(frame) == __frames.end())
104  {
105  Frame new_frame;
106  new_frame.width = tile_width;
107  new_frame.height = tile_height;
108  new_frame.x = (frame % (size_width / tile_width)) * tile_width;
109  new_frame.y = (frame / (size_width / tile_width)) * tile_height;
110  __frames[frame] = new_frame;
