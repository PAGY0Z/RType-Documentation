112  std::vector<Event> WindowManager::GetEvents()
113  {
114  sf::Event event;
115  std::vector<Event> events;
116 
117  while (__window.pollEvent(event))
118  {
119  Event new_event;
120 
121  switch (event.type)
122  {
123  case sf::Event::Closed:
124  new_event.SetClosedEvent();
125  break;
126  case sf::Event::Resized:
127  new_event.SetResizedEvent(event.size.width, event.size.height);
128  break;
129  case sf::Event::LostFocus:
130  new_event.SetLostFocusEvent();
131  break;
132  case sf::Event::GainedFocus:
133  new_event.SetGainedFocusEvent();
134  break;
135  case sf::Event::KeyPressed:
136  new_event.SetKeyPressedEvent(GetKeyboardKeyFromSFML(event.key.code), event.key.alt, event.key.control, event.key.shift, event.key.system);
137  if (new_event.GetKeyEvent().code == EventTypes::KeyboardKey_t::Unknown || new_event.GetKeyEvent().code == EventTypes::KeyboardKey_t::KeyCount)
138  new_event.SetType(EventTypes::EventType_t::Count);
139  break;
140  case sf::Event::KeyReleased:
141  new_event.SetKeyReleasedEvent(GetKeyboardKeyFromSFML(event.key.code), event.key.alt, event.key.control, event.key.shift, event.key.system);
142  if (new_event.GetKeyEvent().code == EventTypes::KeyboardKey_t::Unknown || new_event.GetKeyEvent().code == EventTypes::KeyboardKey_t::KeyCount)
143  new_event.SetType(EventTypes::EventType_t::Count);
144  break;
145  case sf::Event::MouseWheelScrolled:
146  new_event.SetMouseWheelScrolledEvent(GetMouseWheelFromSFML(event.mouseWheelScroll.wheel), event.mouseWheelScroll.delta, event.mouseWheelScroll.x, event.mouseWheelScroll.y);
147  break;
148  case sf::Event::MouseButtonPressed:
149  new_event.SetMouseButtonPressedEvent(GetMouseButtonFromSFML(event.mouseButton.button), event.mouseButton.x, event.mouseButton.y);
150  break;
151  case sf::Event::MouseButtonReleased:
152  new_event.SetMouseButtonReleasedEvent(GetMouseButtonFromSFML(event.mouseButton.button), event.mouseButton.x, event.mouseButton.y);
153  break;
154  case sf::Event::MouseMoved:
155  new_event.SetMouseMovedEvent(event.mouseMove.x, event.mouseMove.y);
156  break;
157  case sf::Event::MouseEntered:
158  new_event.SetMouseEnteredEvent();
159  break;
160  case sf::Event::MouseLeft:
161  new_event.SetMouseLeftEvent();
162  break;
163  default:
164  break;
165  }
166  if (new_event.GetType() != EventTypes::EventType_t::Count)
167  events.push_back(new_event);
168  }
169  return events;
170  }

350  void WindowManager::UpdateShakes(double delta_time)
351  {
352  sf::View view(sf::FloatRect(0, 0, 1920, 1080));
353 
354  view.setSize(1920.0f, 1080.0f);
355  view.setViewport(sf::FloatRect(0, 0, 1, 1));
356 
357  sf::Vector2f offset = {0.f, 0.f};
358 
359  for (auto it = __shakes.begin(); it != __shakes.end();)
360  {
361  if (it->GetDuration() > 0)
362  {
363  double offset_x = (static_cast<double>(rand() % 100) / 100.f - 0.5f) * it->GetIntensity();
364  double offset_y = (static_cast<double>(rand() % 100) / 100.f - 0.5f) * it->GetIntensity();
365 
366  offset.x += offset_x;
367  offset.y += offset_y;
368 
369  it->SetDuration(it->GetDuration() - delta_time);
370 
371  ++it;
372  }
373  else
374  {
375  it = __shakes.erase(it);
376  }
377  }
378  view.move(offset);
379  __window.setView(view);
380  }
